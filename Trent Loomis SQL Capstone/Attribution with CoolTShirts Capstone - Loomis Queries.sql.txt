--Slide 4 – 1.1 How many campaigns and sources does CoolTShirts user and how are they related?SELECT COUNT (DISTINCT utm_campaign)FROM page_visits;SELECT COUNT (DISTINCT utm_source)FROM page_visits;SELECT DISTINCT utm_campaign, utm_sourceFROM page_visits;--Slide 6 – 1.3 What pages are on their website?SELECT DISTINCT page_nameFROM page_visits;--Slide 8 – 2.1 How many first touches is each campaign responsible for?WITH first_touch AS (    SELECT user_id,            MIN(timestamp) as 'first_touch_at'    FROM page_visits    GROUP BY user_id),ft_attr AS (SELECT ft.user_id,              ft.first_touch_at,              pv.utm_source,              pv.utm_campaignFROM first_touch ftJOIN page_visits pv    ON ft.user_id = pv.user_id    AND ft.first_touch_at = pv.timestamp)SELECT ft_attr.utm_source,          ft_attr.utm_campaign,          COUNT(*)FROM ft_attrGROUP BY 1, 2ORDER BY 3 DESC;--Slide 9 – 2.3 How many last touches is each campaign responsible for?WITH last_touch AS (     SELECT user_id,       MAX(timestamp) as 'last_touch_at'            FROM page_visits     GROUP BY user_id),lt_attr AS (SELECT lt.user_id,       lt.last_touch_at,       pv.utm_source,       pv.utm_campaign,       pv.page_nameFROM last_touch ltJOIN page_visits pv    ON lt.user_id = pv.user_id  AND lt.last_touch_at = pv.timestamp)SELECT lt_attr.utm_source,           lt_attr.utm_campaign,     COUNT(*)FROM lt_attrGROUP BY 1, 2ORDER BY 3 DESC;--Slide 10 – 2.3 How many visitors made a purchase?SELECT COUNT (DISTINCT user_id)FROM page_visitsWHERE page_name = '4 - purchase';--Slide 11 – 2.4 How many last touches on the purchase page is each campaign responsible for?WITH last_touch AS (SELECT utm_campaign, user_id,       MAX(timestamp) AS last_touch_atFROM page_visitsWHERE page_name = '4 - purchase'GROUP BY user_id)SELECT utm_campaign, COUNT(*)FROM last_touchGROUP BY utm_campaign;
